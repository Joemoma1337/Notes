#add admin user oneliner
	useradd -m [user] -G sudo && echo "[user]:[pass]"
#Add firewall rule
	sudo ufw allow 80/tcp
#connect to wireless network
	nmcli dev wifi connect "SSID_NAME" password "PASSWORD"
#Delete firewall rule
	sudo ufw delete allow 80/tcp
#Delete folder
	rm -rf /path/to/folder
#disconnect from wireless network
	nmcli dev disconnect wlan0
#export (to .xml) networks and passwords
	sudo cp -r /etc/NetworkManager/system-connections/ .
#FW rule to block 192.168.0.0/16
	sudo ufw deny from 192.168.0.0/16
#FW rule to block 192.168.0.0/16 (internet)
	sudo ufw deny out to 192.168.0.0/16
#Get file hash
	sha256sum /path/to/file
#list all connections
	ss -atu or netstat -an
#list available wireless networks
	nmcli dev wifi
#list available wireless networks (with strength)
	nmcli -f SSID,BARS dev wifi
#list connected storage
	lsblk or df -h
#list connected USB
	lsusb
#List known wireless networks
	nmcli connection show
#list listening connection
	ss -tulpn
#list storage
	sudo fdisk -l
#log off a user
	sudo pkill -u [username]
#query connected wireless network
	nmcli -d or iwconfig
#query event logs and filter by keyword
	journalctl
#query event logs and filter by multiple keywords
	journalctl
#query installed programs
	dpkg-query -l (Debian) or rpm -qa (RHEL)
#query installed programs and filter by name
	dpkg -l
#query locally connected devices
	lspci and lsusb
#query logged on user
	who or w
#Query MAC address vendor
	ip link show (Then check OUI online)
#query public IP
	curl ifconfig.me
#search all directories for a keyword
	grep -rnw "/" -e "keyword"
#show firewall rules
	sudo ufw status numbered
#show firewall rules for specific profile
	sudo ufw status verbose
#show firewall state
	sudo ufw status
