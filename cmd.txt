#Get File Hash
	certutil -hashfile "C:\path\to\file" SHA256
#Query Public IP
	curl ifconfig.me or powershell -command "Invoke-RestMethod -Uri 'http://ipinfo.io/ip'"
#List Storage
	diskpart (then type list disk) or wmic diskdrive get model,size
#Search All Directories (Keyword)
	findstr /s /i /m "keyword" *.*
#Query MAC Address Vendor
	getmac /v (Then manual lookup via OUI)
#Log Off a User
	logoff [SessionID] (Find ID via query user)
#Admin User (One-liner)
	net user [username] [password] /add && net localgroup administrators [username] /add
#FW Rule: Block 192.168.0.0/16 (Internet)
	netsh advfirewall firewall add rule name="BlockOut" dir=out action=block remoteip=192.168.0.0/16
#FW Rule: Block 192.168.0.0/16
	netsh advfirewall firewall add rule name="BlockSubnet" dir=in action=block remoteip=192.168.0.0/16
#Add Firewall Rule
	netsh advfirewall firewall add rule name="MyRule" dir=in action=allow protocol=TCP localport=80
#Delete Firewall Rule
	netsh advfirewall firewall delete rule name="MyRule"
#Show Firewall Rules
	netsh advfirewall firewall show rule name=all
#Show Firewall State
	netsh advfirewall show allprofiles state
#Show Firewall Rules (Profile)
	netsh advfirewall show currentprofile
#Connect to Wireless Network
	netsh wlan connect name="SSID_NAME"
#Disconnect from Wireless
	netsh wlan disconnect
#Export Networks & Passwords
	netsh wlan export profile key=clear folder=C:\
#Query Connected Wireless Network
	netsh wlan show interfaces
#List Available Wireless Networks
	netsh wlan show networks
#List Wireless Networks (with Strength)
	netsh wlan show networks mode=bssid
#List Known Wireless Networks
	netsh wlan show profiles
#List All Connections
	netstat -ano
#List Listening Connections
	netstat -ano
#Query Locally Connected Devices
	pnputil /enum-devices
#Query Logged On User
	query user
#Delete Folder
	rmdir /s /q "C:\Path\To\Folder"
#Query Event Logs (Keyword)
	wevtutil qe System /f:text /c:5 /rd:true
#Query Event Logs (Multiple Keywords)
	wevtutil qe System /f:text /rd:true
#List Connected Storage
	wmic logicaldisk get caption,description,freespace,size
#List Connected USB
	wmic path Win32_USBControllerDevice get Dependent
#Query Installed Programs
	wmic product get name,version
#Query Programs (Filter by Name)
	wmic product where "name like '%%Google%%'" get name
