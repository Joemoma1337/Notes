#Admin User (One-liner)
	New-LocalUser -Name User" -Password ("Pass" | ConvertTo-SecureString -AsPlainText -Force) ; Add-LocalGroupMember -Group "Administrators" -Member "User
#Add Firewall Rule
	New-NetFirewallRule -DisplayName "Allow80" -Direction Inbound -Action Allow -Protocol TCP -LocalPort 80
#Connect to Wireless Network
	netsh wlan connect name="SSID_NAME" (PS doesn't have a native 'connect' cmdlet; netsh is used)
#Delete Firewall Rule
	Remove-NetFirewallRule -DisplayName "Allow80"
#Delete Folder
	Remove-Item -Recurse -Force "C:\Path\To\Folder"
#Disconnect from Wireless
	netsh wlan disconnect
#Export Networks & Passwords
	netsh wlan export profile key=clear folder=$env:USERPROFILE\Desktop
#FW Rule: Block 192.168.0.0/16
	New-NetFirewallRule -DisplayName "BlockSubnet" -Direction Inbound -Action Block -RemoteAddress 192.168.0.0/16
#FW Rule: Block 192.168.0.0/16 (Internet)
	New-NetFirewallRule -DisplayName "BlockOut" -Direction Outbound -Action Block -RemoteAddress 192.168.0.0/16
#Get File Hash
	Get-FileHash -Path "C:\path\file.exe" -Algorithm SHA256
#List All Connections
	Get-NetTCPConnection
#List Available Wireless Networks
	Get-NetAdapter | Where-Object {$_.Status -eq "Up"} (Or netsh wlan show networks)
#List Wireless (with Strength)
	(netsh wlan show networks mode=bssid) -match 'SSID|Signal'
#List Connected Storage
	Get-Volume or Get-PSDrive -PSProvider FileSystem
#List Connected USB
	Get-PnpDevice -PresentOnly | Where-Object { $_.InstanceId -match 'USB' }
#List Known Wireless Networks
	Get-NetAdapter -Name "Wi-Fi" | Get-NetIPAddress (Or netsh wlan show profiles)
#List Listening Connections
	Get-NetTCPConnection -State Listen
#List Storage
	Get-Disk
#Log Off a User
	Invoke-CimMethod -Query "Select * from Win32_OperatingSystem" -MethodName Win32Shutdown -Arguments @{Flags=0}
#Query Connected Wireless Network
	Get-NetAdapter -Name "Wi-Fi" | Select-Object InterfaceDescription, Status, LinkSpeed
#Query Event Logs (Keyword)
	Get-WinEvent -LogName System | Where-Object {$_.Message -like "*Keyword*"}
#Query Event Logs (Multi-Keyword)
	Get-WinEvent -LogName System | Where-Object {$_.Message -match "Error|Warning"}
#Query Installed Programs
	Get-Package or Get-ItemProperty HKLM:\Software\Wow6432Node\Microsoft\Windows\CurrentVersion\Uninstall\* | Select-Object DisplayName
#Query Programs (Filter by Name)
	Get-Package -Name "*Google*"
#Query Locally Connected Devices
	Get-PnpDevice -Status OK
#Query Logged On User
	Get-CimInstance -ClassName Win32_ComputerSystem | Select-Object UserName
#Query MAC Address Vendor
	Get-NetAdapter | Select-Object Name, MacAddress
#Query Public IP
	Invoke-RestMethod -Uri "https://api.ipify.org"
#Search All Directories (Keyword)
	Get-ChildItem -Path "C:\" -Recurse -ErrorAction SilentlyContinue | Select-String "Keyword"
#Show Firewall Rules
	Get-NetFirewallRule
#Show Firewall Rules (Profile)
	Get-NetFirewallRule -PolicyStore ActiveStore
#Show Firewall State
	Get-NetFirewallProfile
